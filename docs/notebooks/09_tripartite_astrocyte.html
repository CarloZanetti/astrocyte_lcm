9. Zemke 2026 Tripartite Astrocyte Signature

This notebook evaluates whether the LCM proteomics data contains an enriched tripartite astrocyte signature, using the 18-gene module from Zemke et al. (2026). The same genes were used in that study with Seurat’s AddModuleScore to identify putative tripartite astrocyte clusters. Here we ask whether these proteins are detected, whether their aggregate abundance differs across conditions, and whether the gene set is enriched in each pairwise comparison.

4. GSEA enrichment across comparisons

Ranks proteins by log2FC (from MSstats) per comparison and tests whether the Zemke tripartite gene set is enriched. This directly mirrors the Habib 2020 DAA analysis in notebook 04.

Code 
# Helper: build a named, sorted rank vector from MSstats DE results
prep_ranks <- function(res, dict, comp_label) {
  res %>%
    filter(Label == comp_label, is.finite(log2FC)) %>%
    left_join(dict, by = "Protein") %>%
    filter(!is.na(Gene), Gene != "") %>%
    mutate(Gene = toupper(Gene)) %>%
    group_by(Gene) %>%
    slice_max(order_by = abs(log2FC), n = 1, with_ties = FALSE) %>%
    ungroup() %>%
    { setNames(.$log2FC, .$Gene) } %>%
    sort(decreasing = TRUE)
}

comparisons <- list(
  "PlaqueNear_vs_Control"   = prep_ranks(res_df, protein_dict, "PlaqueNear_vs_Control"),
  "PlaqueFar_vs_Control"    = prep_ranks(res_df, protein_dict, "PlaqueFar_vs_Control"),
  "PlaqueNear_vs_PlaqueFar" = prep_ranks(res_df, protein_dict, "PlaqueNear_vs_PlaqueFar")

5a. Permutation test: are tripartite genes more abundant than expected by chance?

Tests whether the tripartite gene set has higher absolute protein levels than would be expected if you picked the same number of proteins at random from the full proteome. For each condition the observed mean log-intensity of the detected tripartite genes is compared to a null distribution from 10,000 random draws of equal size. A complementary two-sample Wilcoxon rank-sum test (tripartite vs. all other proteins) is also reported.

Note: this test asks about baseline expression level, not directionality across conditions. See section 5b for the differential enrichment test.

Code 
set.seed(42)
n_perm <- 10000

# Condition-level mean abundance per protein (average across runs within condition)
protein_means_per_cond <- processed_data$ProteinLevelData %>%
  rename(Condition = GROUP) %>%
  left_join(protein_dict, by = "Protein") %>%
  filter(!is.na(Gene) & Gene != "") %>%
  mutate(Gene_upper = toupper(Gene)) %>%
  group_by(Condition, Gene_upper) %>%
  summarise(mean_abundance = mean(LogIntensities, na.rm = TRUE), .groups = "drop") %>%
  filter(!is.nan(mean_abundance))

perm_results_abund <- purrr::map_dfr(cond_levels, function(cond) {

  cond_data <- protein_means_per_cond %>% filter(Condition == cond)
  all_abund  <- cond_data$mean_abundance
  gene_names <- cond_data$Gene_upper

  detected_in_cond <- intersect(detected_zemke, gene_names)
  n_detected        <- length(detected_in_cond)

  if (n_detected < 3) {
    message(glue("Skipping {cond}: fewer than 3 tripartite genes detected"))
    return(NULL)
  }

  observed_mean <- mean(all_abund[gene_names %in% detected_in_cond], na.rm = TRUE)

  null_dist <- replicate(n_perm, mean(sample(all_abund, n_detected, replace = FALSE)))
  emp_pval  <- mean(null_dist >= observed_mean)

  # Two-sample Wilcoxon: tripartite vs. rest of proteome
  tripartite_abund <- all_abund[gene_names %in% detected_in_cond]
  rest_abund       <- all_abund[!(gene_names %in% detected_in_cond)]
  wx <- wilcox.test(tripartite_abund, rest_abund, alternative = "greater")

  tibble(
    Condition     = cond,
    n_genes       = n_detected,
    observed_mean = observed_mean,
    null_mean     = mean(null_dist),
    null_sd       = sd(null_dist),
    perm_pval     = emp_pval,
    wilcox_pval   = wx$p.value,
    null_dist     = list(null_dist)
  )
})

5b. Permutation test: are tripartite genes enriched in abundance across conditions?

This test asks whether the tripartite genes are systematically upregulated in each pairwise comparison — i.e., do they carry consistently positive log2FCs more than a random gene set of the same size would? For each comparison the observed mean log2FC of the tripartite gene set is tested against a null distribution from 10,000 random draws.

A one-sample Wilcoxon signed-rank test (H₀: median log2FC = 0; H₁: > 0) provides a complementary parametric-free test of directional abundance enrichment.

Code 
perm_results_fc <- purrr::map_dfr(names(comparisons), function(comp_name) {

  ranks_vec <- comparisons[[comp_name]]  # named log2FC vector for all proteins

  # Tripartite genes that have an FC value in this comparison
  tripartite_fc <- ranks_vec[names(ranks_vec) %in% zemke_genes]
  n_fc <- length(tripartite_fc)

  if (n_fc < 3) {
    message(glue("Skipping {comp_name}: fewer than 3 tripartite genes with FC data"))
    return(NULL)
  }

  observed_mean_fc <- mean(tripartite_fc)

  # Null: 10,000 random draws of same size from all protein FCs
  null_fc  <- replicate(n_perm, mean(sample(ranks_vec, n_fc, replace = FALSE)))
  emp_pval <- mean(null_fc >= observed_mean_fc)

  # One-sample Wilcoxon signed-rank: is median log2FC of tripartite genes > 0?
  wx_signed <- wilcox.test(tripartite_fc, mu = 0, alternative = "greater")

  tibble(
    Comparison       = comp_name,
    n_genes          = n_fc,
    observed_mean_fc = observed_mean_fc,
    null_mean_fc     = mean(null_fc),
    null_sd_fc       = sd(null_fc),
    perm_pval        = emp_pval,
    wilcox_signed_pval = wx_signed$p.value,
    null_dist        = list(null_fc),
    fc_values        = list(tripartite_fc)
  )
})

perm_results_fc %>%
  dplyr::select(-null_dist, -fc_values) %>%
  mutate(across(where(is.numeric), ~ round(.x, 4)))

6. Cell-type identity validation: astrocyte vs. non-astrocyte marker enrichment

To validate that the LCM samples are genuinely astrocyte-enriched, the same permutation test from section 5a is run across five cell-type marker sets simultaneously. The prediction is that astrocyte marker sets (canonical astrocyte + Zemke tripartite) should have significantly higher-than-chance abundance in the proteome, while neuronal, microglial, and oligodendrocyte marker sets should not. A significant result for non-astrocyte sets would indicate contamination from those cell types.

ct_perm_results <- purrr::map_dfr(names(cell_type_markers), function(ct) {
  genes_in_set    <- cell_type_markers[[ct]]
  detected_in_set <- intersect(genes_in_set, all_genes_pooled)
  n_det           <- length(detected_in_set)

  if (n_det < 3) {
    message(glue("Skipping {ct}: fewer than 3 genes detected"))
    return(NULL)
  }

  observed_mean <- mean(all_abund_pooled[all_genes_pooled %in% detected_in_set])

  null_dist <- replicate(n_perm,
                         mean(sample(all_abund_pooled, n_det, replace = FALSE)))
  emp_pval  <- mean(null_dist >= observed_mean)

  # Effect size: how many SDs above the null mean (z-score equivalent)
  effect_size <- (observed_mean - mean(null_dist)) / sd(null_dist)

  # Two-sample Wilcoxon: this cell type vs. all other proteins
  target_abund <- all_abund_pooled[all_genes_pooled %in% detected_in_set]
  rest_abund   <- all_abund_pooled[!(all_genes_pooled %in% detected_in_set)]
  wx <- wilcox.test(target_abund, rest_abund, alternative = "greater")

  tibble(
    Cell_type     = ct,
    n_detected    = n_det,
    observed_mean = observed_mean,
    null_mean     = mean(null_dist),
    null_sd       = sd(null_dist),
    effect_size_z = effect_size,
    perm_pval     = emp_pval,
    wilcox_pval   = wx$p.value,
    null_dist     = list(null_dist)
  )
})
